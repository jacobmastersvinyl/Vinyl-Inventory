// ═══════════════════════════════════════════════════════════════ // FUNKIFIED — Job Packing extension // Lets the Funkified PWA "Pack a client order" against a pending // portal order (e.g. NRMAi). Scanning ticks items off the packing // list, decrements the Inventory sheet, stamps the order rows, and // logs each scan to the Scan Log. // // Paste this as a NEW file inside the same Apps Script project as // Code.gs and ClientPortal.gs: // Apps Script editor → + (next to Files) → Script → name it // "JobPacking". // // Globals from Code.gs (CONFIG, ss(), getSheet(), INV_COLS, etc.) // are in scope automatically. // // Dispatch (wire these two snippets into Code.gs — see bottom): // POST {action:'jobsPending', team:'Jacob'} // POST {action:'jobPack', orderId:'NRMAi-20260421-…'} // POST {action:'jobComplete', orderId:'NRMAi-…', scans:[{shelf,qty,extra}], closeOut:true|false} // ═══════════════════════════════════════════════════════════════ const JOB_CONFIG = { // Which clients to scan for jobs. Each must have a // " Orders" sheet managed by ClientPortal.gs. CLIENTS: ['NRMAi'], // Statuses that should appear in the warehouse job picker. // Pending orders are still being edited by the client and aren't // ready to pack yet — only Scheduled (quoted + agreed) and Partial // (already started) jobs show up here. PICKABLE_STATUSES: ['Scheduled', 'Partial'], // Terminal status when every line is packed in full. STATUS_ON_SITE: 'On-site', // Status when we close a job with shortfall. STATUS_PARTIAL: 'Partial', // Columns we add to the Orders sheet if not already present. // These are appended to whatever ORDER_HEADER ClientPortal.gs wrote. EXTRA_HEADERS: ['Packed Qty', 'Packed By', 'Packed At'] }; // ─────────── Dispatch shims (called from Code.gs) ─────────── function jobPackGet(action, params) { // Currently no GET endpoints — keep the hook for future use. return null; } function jobPackPost(action, body) { if (action === 'jobspending') return jsonOut(listPendingJobs_(body)); if (action === 'jobpack') return jsonOut(getJobPack_(body)); if (action === 'jobcomplete') return jsonOut(completeJobPack_(body)); return null; } // ─────────── Sheet helpers specific to Orders sheets ─────────── function orderSheetName_(client) { return client + ' Orders'; } // Build a { headerName -> 1-based column index } map from the sheet header row. function headerMap_(sheet) { const last = sheet.getLastColumn(); if (last < 1) return {}; const row = sheet.getRange(1, 1, 1, last).getValues()[0]; const map = {}; for (let i = 0; i < row.length; i++) { const k = String(row[i] || '').trim(); if (k) map[k] = i + 1; } return map; } // Make sure the extra packing columns exist. Returns the up-to-date header map. function ensurePackingHeaders_(sheet) { let map = headerMap_(sheet); let nextCol = sheet.getLastColumn() + 1; for (let i = 0; i < JOB_CONFIG.EXTRA_HEADERS.length; i++) { const h = JOB_CONFIG.EXTRA_HEADERS[i]; if (!map[h]) { sheet.getRange(1, nextCol).setValue(h).setFontWeight('bold'); map[h] = nextCol; nextCol++; } } sheet.setFrozenRows(1); return map; } // Read all rows of an Orders sheet with a header-name lookup. // Returns { headers, rows: [{ _row, Order ID, ... }] }. function readOrders_(sheet) { const last = sheet.getLastRow(); if (last < 2) return { headers: headerMap_(sheet), rows: [] }; const width = sheet.getLastColumn(); const values = sheet.getRange(2, 1, last - 1, width).getValues(); const headers = headerMap_(sheet); const rows = []; for (let i = 0; i < values.length; i++) { const o = { _row: i + 2 }; for (const k in headers) { o[k] = values[i][headers[k] - 1]; } rows.push(o); } return { headers, rows }; } // Pull the display-friendly drop-off date and address fields from an order row, // regardless of whether the sheet uses the original "Delivered By/Picked Up By/ // Drop-off Address" schema or the newer schedule schema ("Drop-off Date/Time / // Pick-up Date/Time / Delivery Address / Access Notes"). function extractMeta_(row) { const dropoffWhen = row['Drop-off Date/Time'] || row['Drop-off Date'] || ''; const pickupWhen = row['Pick-up Date/Time'] || row['Pick-up Date'] || ''; const address = row['Delivery Address'] || row['Drop-off Address'] || ''; const notes = row['Access Notes'] || row['Notes'] || ''; const deliveredBy = row['Delivered By'] || ''; const pickedUpBy = row['Picked Up By'] || ''; return { dropoffWhen: dropoffWhen instanceof Date ? dropoffWhen.toISOString() : String(dropoffWhen || ''), pickupWhen: pickupWhen instanceof Date ? pickupWhen.toISOString() : String(pickupWhen || ''), address: String(address || ''), notes: String(notes || ''), deliveredBy: String(deliveredBy || ''), pickedUpBy: String(pickedUpBy || '') }; } // ─────────── Action: list pending jobs ─────────── function listPendingJobs_(body) { const jobs = []; for (let c = 0; c < JOB_CONFIG.CLIENTS.length; c++) { const client = JOB_CONFIG.CLIENTS[c]; const sheet = ss().getSheetByName(orderSheetName_(client)); if (!sheet) continue; const { rows } = readOrders_(sheet); // Group rows by Order ID so each job is a single picker item. const byOrder = {}; for (let i = 0; i < rows.length; i++) { const r = rows[i]; const status = String(r['Status'] || '').trim(); if (JOB_CONFIG.PICKABLE_STATUSES.indexOf(status) < 0) continue; const id = String(r['Order ID'] || '').trim(); if (!id) continue; if (!byOrder[id]) { byOrder[id] = { orderId: id, client, status, jobName: String(r['Job Name'] || '').trim(), placedBy: String(r['Placed By'] || '').trim(), createdAt: r['Timestamp'] instanceof Date ? r['Timestamp'].toISOString() : String(r['Timestamp'] || ''), meta: extractMeta_(r), lineCount: 0, unitsOrdered: 0, unitsPacked: 0 }; } const qty = Number(r['Qty']) || 0; const packed = Number(r['Packed Qty']) || 0; byOrder[id].lineCount++; byOrder[id].unitsOrdered += qty; byOrder[id].unitsPacked += packed; } for (const id in byOrder) { const j = byOrder[id]; j.unitsOutstanding = Math.max(0, j.unitsOrdered - j.unitsPacked); jobs.push(j); } } // Sort by drop-off date ascending (closest delivery first), then by // status so Scheduled appears before Partial. jobs.sort(function (a, b) { const ad = Date.parse(a.meta.dropoffWhen) || 0; const bd = Date.parse(b.meta.dropoffWhen) || 0; if (ad !== bd) return ad - bd; if (a.status !== b.status) return a.status === 'Scheduled' ? -1 : 1; return String(a.createdAt).localeCompare(String(b.createdAt)); }); return { success: true, jobs }; } // ─────────── Action: get a single job's pack-list ─────────── function getJobPack_(body) { const orderId = String(body.orderId || '').trim(); if (!orderId) return { success: false, error: 'orderId required' }; // Find which client sheet the order is on. let sheet = null, rowsForOrder = []; for (let c = 0; c < JOB_CONFIG.CLIENTS.length; c++) { const client = JOB_CONFIG.CLIENTS[c]; const s = ss().getSheetByName(orderSheetName_(client)); if (!s) continue; const { rows } = readOrders_(s); const match = rows.filter(function (r) { return String(r['Order ID'] || '').trim() === orderId; }); if (match.length) { sheet = s; rowsForOrder = match; break; } } if (!sheet || !rowsForOrder.length) return { success: false, error: 'Order not found: ' + orderId }; const inv = getSheet(CONFIG.SHEET_INVENTORY); ensureHeaders(inv, INV_HEADER); const head = rowsForOrder[0]; const meta = extractMeta_(head); // Build the packing lines. Look up current available for each shelf. const items = rowsForOrder.map(function (r) { const shelf = normShelf(r['Shelf']); const qty = Number(r['Qty']) || 0; const packed = Number(r['Packed Qty']) || 0; let available = null, invName = '', totalQty = 0, qtyOut = 0; if (shelf) { const rIdx = findRowByShelf(inv, shelf); if (rIdx > 0) { const cur = readRow(inv, rIdx); invName = cur.name; totalQty = Number(cur.totalQty) || 0; qtyOut = Number(cur.qtyOut) || 0; available = Math.max(0, totalQty - qtyOut); } } return { shelf, itemName: String(r['Item Name'] || invName || ''), qtyOrdered: qty, qtyPacked: packed, qtyOutstanding: Math.max(0, qty - packed), available, // null if shelf unknown / not registered yet totalQty, qtyOut }; }); return { success: true, orderId, client: String(head['Client'] || '').trim(), status: String(head['Status'] || '').trim(), jobName: String(head['Job Name'] || '').trim(), placedBy: String(head['Placed By'] || '').trim(), createdAt: head['Timestamp'] instanceof Date ? head['Timestamp'].toISOString() : String(head['Timestamp'] || ''), meta, items }; } // ─────────── Action: complete (scan off) a job's items ─────────── // // body.scans: [{ shelf, qty, extra? }] // - shelf is the key; qty is how many to pack in THIS call // - extra=true means it isn't on the packing list (still decrements // inventory + logs, but doesn't count against any order line) // body.closeOut: if true, set the order's final status now: // - On-site if every line fully packed // - Partial otherwise (stays pickable — you can finish later) // function completeJobPack_(body) { const orderId = String(body.orderId || '').trim(); if (!orderId) return { success: false, error: 'orderId required' }; const scans = Array.isArray(body.scans) ? body.scans : []; if (!scans.length) return { success: false, error: 'No scans submitted' }; // Find the order sheet and rows. let sheet = null, rowMap = {}, client = ''; for (let c = 0; c < JOB_CONFIG.CLIENTS.length; c++) { const s = ss().getSheetByName(orderSheetName_(JOB_CONFIG.CLIENTS[c])); if (!s) continue; const { rows } = readOrders_(s); const match = rows.filter(function (r) { return String(r['Order ID'] || '').trim() === orderId; }); if (match.length) { sheet = s; client = JOB_CONFIG.CLIENTS[c]; // Map shelf -> row(s). Preserve order so first-come, first-served. match.forEach(function (r) { const shelf = normShelf(r['Shelf']); if (!rowMap[shelf]) rowMap[shelf] = []; rowMap[shelf].push(r); }); break; } } if (!sheet) return { success: false, error: 'Order not found: ' + orderId }; const headers = ensurePackingHeaders_(sheet); const inv = getSheet(CONFIG.SHEET_INVENTORY); ensureHeaders(inv, INV_HEADER); const team = String(body.team || '').trim(); const now = nowDate(); const applied = []; // { shelf, qtyApplied, extra, shortOfRequest } const warnings = []; for (let i = 0; i < scans.length; i++) { const scan = scans[i] || {}; const shelf = normShelf(scan.shelf); let qty = Math.max(1, Number(scan.qty) || 1); const extra = !!scan.extra; if (!shelf) { warnings.push({ reason: 'blank-shelf', scan }); continue; } // Decrement inventory (mirrors actTakeOut behaviour). const rIdx = findRowByShelf(inv, shelf); if (rIdx < 0) { warnings.push({ shelf, reason: 'shelf-not-in-inventory' }); continue; } const cur = readRow(inv, rIdx); const total = Number(cur.totalQty) || 0; const curOut = Number(cur.qtyOut) || 0; const availableNow = Math.max(0, total - curOut); const willTake = total ? Math.min(qty, availableNow) : qty; if (willTake <= 0) { warnings.push({ shelf, name: cur.name, reason: 'out-of-stock' }); continue; } const newOut = total ? Math.min(total, curOut + willTake) : curOut + willTake; inv.getRange(rIdx, INV_COLS.QtyOut).setValue(newOut); inv.getRange(rIdx, INV_COLS.Status) .setValue(total && newOut >= total ? 'out' : (newOut > 0 ? 'partial' : 'in')); inv.getRange(rIdx, INV_COLS.LastScanned).setValue(now); inv.getRange(rIdx, INV_COLS.LastScannedBy).setValue(team); // Allocate to order lines by shelf (fill earliest outstanding lines first). let remaining = willTake; const linesForShelf = rowMap[shelf] || []; if (!extra) { for (let j = 0; j < linesForShelf.length && remaining > 0; j++) { const line = linesForShelf[j]; const ordered = Number(line['Qty']) || 0; const already = Number(line['Packed Qty']) || 0; const room = Math.max(0, ordered - already); if (room <= 0) continue; const take = Math.min(room, remaining); const newPacked = already + take; sheet.getRange(line._row, headers['Packed Qty']).setValue(newPacked); sheet.getRange(line._row, headers['Packed By']).setValue(team); sheet.getRange(line._row, headers['Packed At']).setValue(now); // Keep this in-memory too so subsequent scans against the same shelf see it. line['Packed Qty'] = newPacked; remaining -= take; } } // If there's leftover qty and it wasn't marked extra, flag it. if (remaining > 0 && !extra) { warnings.push({ shelf, name: cur.name, reason: 'over-packed', over: remaining }); } appendScan( { team, gpsLat: scan.gpsLat, gpsLng: scan.gpsLng, locationLabel: scan.locationLabel }, shelf, cur.name, extra ? 'PACK (EXTRA): ' + orderId + ' ×' + willTake : 'PACK: ' + orderId + ' ×' + willTake, orderId ); applied.push({ shelf, name: cur.name, qtyApplied: willTake, qtyRequested: qty, extra, shortOfRequest: qty - willTake }); } // Recompute order status after applying scans. const fresh = readOrders_(sheet).rows.filter(function (r) { return String(r['Order ID'] || '').trim() === orderId; }); const allPacked = fresh.every(function (r) { const ordered = Number(r['Qty']) || 0; const packed = Number(r['Packed Qty']) || 0; return packed >= ordered; }); const anyPacked = fresh.some(function (r) { return (Number(r['Packed Qty']) || 0) > 0; }); let newStatus = null; if (body.closeOut) { newStatus = allPacked ? JOB_CONFIG.STATUS_ON_SITE : JOB_CONFIG.STATUS_PARTIAL; } else if (allPacked) { // Even without closeOut, if every line is fully packed we're On-site. newStatus = JOB_CONFIG.STATUS_ON_SITE; } else if (anyPacked) { newStatus = JOB_CONFIG.STATUS_PARTIAL; } if (newStatus) { const statusCol = headers['Status']; for (let i = 0; i < fresh.length; i++) { sheet.getRange(fresh[i]._row, statusCol).setValue(newStatus); } } return { success: true, orderId, client, applied, warnings, status: newStatus || (fresh[0] && String(fresh[0]['Status'] || '')) }; }
